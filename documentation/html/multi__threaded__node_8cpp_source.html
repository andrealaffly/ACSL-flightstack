130  // Each of the callback groups is basically a thread

132  auto sub_pixhawk_odometry_opt = rclcpp::SubscriptionOptions();
133  sub_pixhawk_odometry_opt.callback_group = callback_group_sub_pixhawk_odometry_;

135  // Quality Of Service initialization for "subscription_pixhawk_odometry_"
136  rmw_qos_profile_t qos_profile = rmw_qos_profile_sensor_data;
137  auto qos = rclcpp::QoS(rclcpp::QoSInitialization(qos_profile.history, 5), qos_profile);

139  // Creation of the Pixhawk vehicle odometry subscription

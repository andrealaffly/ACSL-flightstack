174 // Function that computes the Jacobian matrix given roll and pitch angles
175 Eigen::Matrix3d Control::jacobianMatrix(const double& roll, const double& pitch)
176 {
177  Eigen::Matrix3d jacobian_matrix;
178 
179  jacobian_matrix(0, 0) = 1;
180  jacobian_matrix(0, 1) = 0;
181  jacobian_matrix(0, 2) = -std::sin(pitch);
182 
183  jacobian_matrix(1, 0) = 0;
184  jacobian_matrix(1, 1) = std::cos(roll);
185  jacobian_matrix(1, 2) = std::sin(roll) * std::cos(pitch);
186 
187  jacobian_matrix(2, 0) = 0;
188  jacobian_matrix(2, 1) = -std::sin(roll);
189  jacobian_matrix(2, 2) = std::cos(roll) * std::cos(pitch);
190 
191  return jacobian_matrix;
192 }

196 Eigen::Matrix3d Control::jacobianMatrixDot(const double& roll,
197  const double& pitch,
198  const double& roll_dot,
199  const double& pitch_dot)

203  jacobian_matrix_dot(0, 0) = 0;

206 
207  jacobian_matrix_dot(1, 0) = 0;
208  jacobian_matrix_dot(1, 1) = -std::sin(roll) * roll_dot;
209  jacobian_matrix_dot(1, 2) = std::cos(roll) * std::cos(pitch) * roll_dot
210  - std::sin(roll) * std::sin(pitch) * pitch_dot;
211 
212  jacobian_matrix_dot(2, 0) = 0;
213  jacobian_matrix_dot(2, 1) = -std::cos(roll) * roll_dot;
214  jacobian_matrix_dot(2, 2) = -std::sin(roll) * std::cos(pitch) * roll_dot
215  - std::cos(roll) * std::sin(pitch) * pitch_dot;
216 
217  return jacobian_matrix_dot;

220 // Function that returns a 3x3 Jacobian matrix inverse given roll and pitch angles.
221 Eigen::Matrix3d Control::jacobianMatrixInverse(const double& roll, const double& pitch)
222 {
223  // Ensure pitch is not near +/- pi/2 to avoid division by zero
224  if (std::abs(std::cos(pitch)) < 1e-6) {
225  throw std::invalid_argument("Pitch value results in division by zero.");
226  }
227 
228  Eigen::Matrix3d jacobian_inverse;
229 
230  jacobian_inverse(0, 0) = 1;
231  jacobian_inverse(0, 1) = (std::sin(roll) * std::sin(pitch)) / std::cos(pitch);
232  jacobian_inverse(0, 2) = (std::cos(roll) * std::sin(pitch)) / std::cos(pitch);
233 
234  jacobian_inverse(1, 0) = 0;
235  jacobian_inverse(1, 1) = std::cos(roll);
236  jacobian_inverse(1, 2) = -std::sin(roll);
237 
238  jacobian_inverse(2, 0) = 0;
239  jacobian_inverse(2, 1) = std::sin(roll) / std::cos(pitch);
240  jacobian_inverse(2, 2) = std::cos(roll) / std::cos(pitch);
241 
242  return jacobian_inverse;
243 }

245 /*
246  Function that computes the 321 rotation matrix that transforms a vector from global to local coordinates.
247 
248  rotation_matrix_321_local_to_global = R3 * R2 * R1; // hence
249  rotation_matrix_321_global_to_local = R1_transpose * R2_transpose * R3_transpose;
250 */
251 Eigen::Matrix3d Control::rotationMatrix321GlobalToLocal(const double& roll, const double& pitch, const double& yaw)
252 {
253  Eigen::Matrix3d rotation_matrix_321_global_to_local;
254  Eigen::Matrix3d R1_transpose;
255  Eigen::Matrix3d R2_transpose;
256  Eigen::Matrix3d R3_transpose;
257 
258  R1_transpose = (Eigen::Matrix3d() <<
259  1, 0, 0,
260  0, std::cos(roll), std::sin(roll),
261  0, -std::sin(roll), std::cos(roll)
262  ).finished();
263 
264  R2_transpose = (Eigen::Matrix3d() <<
265  std::cos(pitch), 0, -std::sin(pitch),
266  0, 1, 0,
267  std::sin(pitch), 0, std::cos(pitch)
268  ).finished();
269 
270  R3_transpose = (Eigen::Matrix3d() <<
271  std::cos(yaw), std::sin(yaw), 0,
272  -std::sin(yaw), std::cos(yaw), 0,
273  0, 0, 1
274  ).finished();
275 
276  rotation_matrix_321_global_to_local = R1_transpose * R2_transpose * R3_transpose;
277 
278  return rotation_matrix_321_global_to_local;
279 }

377 /*
378  Function that computes:
379  the Total Thrust U1,
380  the desired roll angle RollDes,
381  the desired pitch angle PitchDes.
382  U1 will be fed to mixer matrix, while RollDes and PitchDes will be fed to the inner loop.
383 */
384 void Control::compute_U1_RollDes_PitchDes(ControlInternalMembers& cim)
385 {
386  // Convert the mu_translational from global to local coordinates
387  cim.mu_translational = cim.rotation_matrix_321_global_to_local * cim.mu_translational_raw;
388 
389  cim.U_control_inputs[0] = std::sqrt(
390  std::pow(cim.mu_translational[0], 2)
391  + std::pow(cim.mu_translational[1], 2)
392  + std::pow(cim.mu_translational[2], 2)
393  );
394 
395  double temporaryvar_roll_desired_1 = cim.mu_translational[1] / cim.U_control_inputs[0];
396  cim.roll_desired= std::atan2(
397  temporaryvar_roll_desired_1,
398  std::sqrt(1 - std::pow(temporaryvar_roll_desired_1, 2))
399  );
400 
401  cim.pitch_desired = std::atan2(-cim.mu_translational[0], -cim.mu_translational[2]);
402 }

468 /*
469  Function that makes the yaw angular error continuos considering the discontinuity/wrapping that occurs
470  at +/- pi for both the yaw and the user-defined yaw
471 */
472 double Control::makeYawAngularErrorContinuous(double yaw, double user_defined_yaw)
473 {
474  double continuos_error;
475 
476  user_defined_yaw = wrapAngleToMinusPiAndPi(user_defined_yaw);
477 
478  double raw_error = yaw - user_defined_yaw;
479 
480  if ( std::abs(raw_error) >= M_PI && std::abs(raw_error) < 2*M_PI && raw_error < 0 )
481  {
482  continuos_error = std::fmod(raw_error, M_PI) + M_PI;
483  }
484  else if ( std::abs(raw_error) >= M_PI && std::abs(raw_error) < 2*M_PI && raw_error > 0 )
485  {
486  continuos_error = std::fmod(raw_error, M_PI) - M_PI;
487  }
488  else
489  {
490  continuos_error = std::fmod(raw_error, M_PI);
491  }
492 
493  return continuos_error;
494 }

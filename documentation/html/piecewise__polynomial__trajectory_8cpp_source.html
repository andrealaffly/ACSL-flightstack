360 // Function to assign polynomial coefficients to designated variables
367 std::vector<Eigen::MatrixXd> polyCoefAssigning(const std::vector<std::vector<double>>& poly_coeff_matrix_in)
368 {
369  // Get the size of the input matrix
370  int rows = poly_coeff_matrix_in.size();
371  int cols = poly_coeff_matrix_in[0].size();
372 
373  // Calculate the size of the output matrices
374  int out_rows = rows / 3;
375 
376  // Initialize output matrices for x, y, and z variables
377  Eigen::MatrixXd poly_coeff_matrix_out_x(out_rows, cols);
378  Eigen::MatrixXd poly_coeff_matrix_out_y(out_rows, cols);
379  Eigen::MatrixXd poly_coeff_matrix_out_z(out_rows, cols);
380 
381  // Assign polynomial coefficients to designated variables
382  for (int i = 0; i < out_rows; ++i) {
383  for (int j = 0; j < cols; ++j) {
384  poly_coeff_matrix_out_x(i, j) = poly_coeff_matrix_in[3 * i][j];
385  poly_coeff_matrix_out_y(i, j) = poly_coeff_matrix_in[3 * i + 1][j];
386  poly_coeff_matrix_out_z(i, j) = poly_coeff_matrix_in[3 * i + 2][j];
387  }
388  }
389 
390  // Return the output matrices
391  return {poly_coeff_matrix_out_x, poly_coeff_matrix_out_y, poly_coeff_matrix_out_z};
392 }

401 Eigen::MatrixXd polyDerMatrix(const Eigen::MatrixXd& poly_coeff_matrix_in)
402 {
403  // Get the size of the input matrix
404  int rows = poly_coeff_matrix_in.rows();
405  int cols = poly_coeff_matrix_in.cols();
406 
407  // Initialize the output matrix for the derivative
408  Eigen::MatrixXd poly_coeff_matrix_out(rows, cols - 1);
409 
410  // Compute the derivative for each row of the input matrix
411  for (int i = 0; i < rows; ++i) {
412  for (int j = 0; j < cols - 1; ++j) {
413  poly_coeff_matrix_out(i, j) = (cols - 1 - j) * poly_coeff_matrix_in(i, j);
414  }
415  }
416 
417  return poly_coeff_matrix_out;
418 }

524 // Compute the user-defined yaw_dot from the trajectory velocity in the XY plane
535 double yawDotComputation(const Eigen::VectorXd& Vx_coef, const Eigen::VectorXd& Vy_coef,
536  const Eigen::VectorXd& Ax_coef, const Eigen::VectorXd& Ay_coef, double t)
537 {
538  // Compute the acceleration components Ax and Ay at time t using polynomial coefficients
539  double Ax = evaluatePolynomial(Ax_coef, t);
540  double Ay = evaluatePolynomial(Ay_coef, t);
541 
542  // Compute the acceleration angle in the complex plane
543  double acceleration_angle = std::atan2(Ay, Ax);
544 
545  // Compute the velocity and acceleration norms
546  double velocity_norm = norm2D(Vx_coef, Vy_coef, t);
547  double acceleration_norm = norm2D(Ax_coef, Ay_coef, t);
548 
549  // Compute the yaw angle
550  double yaw = yawComputation(Vx_coef, Vy_coef, t);
551 
552  // Compute the yaw rate
553  double yaw_dot = (acceleration_norm / velocity_norm) * std::sin(acceleration_angle - yaw);
554 
555  return yaw_dot;

558 // Compute the user-defined yaw_dot_dot from the trajectory velocity in the XY plane
571 double yawDotDotComputation(const Eigen::VectorXd& Vx_coef, const Eigen::VectorXd& Vy_coef,
572  const Eigen::VectorXd& Ax_coef, const Eigen::VectorXd& Ay_coef,
573  const Eigen::VectorXd& Jx_coef, const Eigen::VectorXd& Jy_coef,
574  double t)

577  double Jx = evaluatePolynomial(Jx_coef, t);

580  double Jy = evaluatePolynomial(Jy_coef, t);

585  // Compute the velocity and jerk norms

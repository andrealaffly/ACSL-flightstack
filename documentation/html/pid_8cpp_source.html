174 /*
175  Function to compute the variables used for the differentiator of the desired roll and pitch angles.
176  This function calculates the internal states and derivatives (first and second) of the desired roll and pitch
177  angles using the provided filter coefficients and the current state variables.
178 */
179 void PID::computeFilterDifferentiatorVariables(ControlInternalMembers& cim,
180  VehicleInfo& vehicle_info,
181  StateController& state_)
182 {
183  cim.internal_state_roll_des_filter = vehicle_info.A_filter_roll_des * state_.state_roll_des_filter
184  + vehicle_info.B_filter_roll_des * cim.roll_desired;
185 
186  cim.internal_state_pitch_des_filter = vehicle_info.A_filter_pitch_des * state_.state_pitch_des_filter
187  + vehicle_info.B_filter_pitch_des * cim.pitch_desired;
188 
189  cim.roll_desired_dot = vehicle_info.C_filter_roll_des * state_.state_roll_des_filter;
190  cim.pitch_desired_dot = vehicle_info.C_filter_pitch_des * state_.state_pitch_des_filter;
191 
192 
193  cim.internal_state_roll_dot_des_filter = vehicle_info.A_filter_roll_dot_des * state_.state_roll_dot_des_filter
194  + vehicle_info.B_filter_pitch_des * cim.roll_desired_dot;
195 
196  cim.internal_state_pitch_dot_des_filter = vehicle_info.A_filter_pitch_dot_des * state_.state_pitch_dot_des_filter
197  + vehicle_info.B_filter_pitch_dot_des * cim.pitch_desired_dot;
198 
199  cim.roll_desired_dot_dot = vehicle_info.C_filter_roll_dot_des * state_.state_roll_dot_des_filter;
200  cim.pitch_desired_dot_dot = vehicle_info.C_filter_pitch_dot_des * state_.state_pitch_dot_des_filter;
201 }

203 /*
204  OUTER LOOP CONTROLLER
205 */
206 void PID::computeOuterLoop(ControlInternalMembers& cim,
207  VehicleInfo& vehicle_info,
208  StateController& state_,
209  ControlReferences& cr,
210  GainsPID& gains_,
211  ControllerSpecificInternalMembers& csim_)
212 {
213  csim_.outer_loop_dynamic_inversion = - vehicle_info.mass * GRAVITATIONAL_ACCELERATION * this->e3_basis;
214  csim_.outer_loop_P = - gains_.KP_translational * cim.translational_position_error;
215  csim_.outer_loop_D = - gains_.KD_translational * (cr.velocity - cr.user_defined_velocity);
216  csim_.outer_loop_I = - gains_.KI_translational * state_.integral_translational_position_error;
217 
218  cim.mu_translational_raw = csim_.outer_loop_dynamic_inversion +
219  vehicle_info.mass * (
220  csim_.outer_loop_P
221  + csim_.outer_loop_D
222  + csim_.outer_loop_I
223  + cr.user_defined_acceleration
224  );
225 }

227 /*
228  INNER LOOP CONTROLLER
229 */
230 void PID::computeInnerLoop(ControlInternalMembers& cim,
231  VehicleInfo& vehicle_info,
232  StateController& state_,
233  ControlReferences& cr,
234  GainsPID& gains_,
235  ControllerSpecificInternalMembers& csim_)
236 {
237  csim_.inner_loop_dynamic_inversion = cr.angular_velocity.cross(vehicle_info.inertia_matrix * cr.angular_velocity);
238  csim_.inner_loop_P = - gains_.KP_rotational * cim.angular_error;
239  csim_.inner_loop_D = - gains_.KD_rotational * (cim.euler_angles_rpy_dot - cim.angular_position_desired_dot);
240  csim_.inner_loop_I = - gains_.KI_rotational * state_.integral_angular_error;
241 
242  cim.U2_U3_U4 = csim_.inner_loop_dynamic_inversion +
243  vehicle_info.inertia_matrix * (
244  csim_.inner_loop_P
245  + csim_.inner_loop_D
246  + csim_.inner_loop_I
247  + cim.angular_position_desired_dot_dot
248  );
249 }

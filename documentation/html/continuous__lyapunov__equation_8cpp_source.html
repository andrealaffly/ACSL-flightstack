124 MatrixXd RealContinuousLyapunovEquation(const Eigen::Ref<const MatrixXd>& A,
125  const Eigen::Ref<const MatrixXd>& Q) {
126  if (A.rows() != A.cols() || Q.rows() != Q.cols() || A.rows() != Q.rows()) {
127  throw std::runtime_error(
128  "RealContinuousLyapunovEquation(): A and Q must be square and of equal "
129  "size!");
130  }
131 
132  if (!is_approx_equal_abstol(Q, Q.transpose(), 1e-10)) {
133  throw std::runtime_error("Q is not symmetric.");
134  }
135 
136  // The cases where A is 1-by-1 or 2-by-2 are caught as we have an efficient
137  // methods at hand to solve these without the use of Schur factorization.
138  if (static_cast<int>(A.cols()) == 1) {
139  if (is_zero(A(0, 0))) {
140  throw std::runtime_error(
141  "RealContinuousLyapunovEquation(): Solution is not unique!");
142  }
143  return internal::Solve1By1RealContinuousLyapunovEquation(A, Q);
144  }
145  if (static_cast<int>(A.cols()) == 2) {
146  // If A has two real eigenvalues λ₁ and λ₂, then we check (1) if λ₁ or λ₂ is
147  // 0, i.e. if the determinant of A is 0; (2) if λ₁ + λ₂ = 0, i.e., if the
148  // trace of A is 0. If A has two complex eigenvalues λ₁ and λ̅₁, then we
149  // check if λ₁ + λ̅₁ = 0, i.e., if the trace of A is 0.
150  if (is_zero(A(0, 0) + A(1, 1)) ||
151  is_zero(A(0, 0) * A(1, 1) - A(1, 0) * A(0, 1))) {
152  throw std::runtime_error(
153  "RealContinuousLyapunovEquation(): Solution is not unique!");
154  }
155  // Reducing Q to upper triangular form.
156  MatrixXd Q_upper{Q};
157  Q_upper(1, 0) = NAN;
158  return internal::Solve2By2RealContinuousLyapunovEquation(A, Q_upper);
159  }
160  VectorXcd eig_val{A.eigenvalues()};
161  for (int i = 0; i < eig_val.size(); ++i) {
162  for (int j = i; j < eig_val.size(); ++j) {
163  std::complex<double> eig_sum = eig_val(i) + std::conj(eig_val(j));
164  if (is_zero(eig_sum.real()) && is_zero(eig_sum.imag())) {
165  throw std::runtime_error(
166  "RealContinuousLyapunovEquation(): Solution is not unique!");
167  }
168  }
169  }
170  // Transform into reduced form S'*X̅ + X̅ *S = -Q̅, where A = U*S*U', X̅ =
171  // U'*X*U, Q̅ = U'*Q*U, U is the unitary matrix, and S the reduced form. Note
172  // that the expression for X̅ in [2] is incorrect.
173  Eigen::RealSchur<MatrixXd> schur(A);
174  const MatrixXd U{schur.matrixU()};
175  const MatrixXd S{schur.matrixT()};
176  if (schur.info() != Eigen::Success) {
177  throw std::runtime_error(
178  "RealContinuousLyapunovEquation(): Schur factorization failed.");
179  }
180  // Reduce the symmetric Q̅ to its upper triangular form Q̅_ᵤₚₚₑᵣ.
181  MatrixXd Q_bar_upper{MatrixXd::Constant(Q.rows(), Q.cols(), NAN)};
182  Q_bar_upper.triangularView<Eigen::Upper>() = U.transpose() * Q * U;
183  return (U *
184  internal::SolveReducedRealContinuousLyapunovEquation(S, Q_bar_upper) *
185  U.transpose());
186 }

199 Matrix2d Solve2By2RealContinuousLyapunovEquation(
200  const Eigen::Ref<const Matrix2d>& A, const Eigen::Ref<const Matrix2d>& Q) {
201  // Rewrite AᵀX + XA + Q = 0 (case where A,Q are 2-by-2) into linear set A_vec
202  // *vec(X) = -vec(Q) and solve for X.
203  // Only the upper triangular part of Q is used, thus it is treated to be
204  // symmetric.
205  Matrix3d A_vec;
206  A_vec << 2 * A(0, 0), 2 * A(1, 0), 0, A(0, 1), A(0, 0) + A(1, 1), A(1, 0), 0,
207  2 * A(0, 1), 2 * A(1, 1);
208  const Vector3d Q_vec(-Q(0, 0), -Q(0, 1), -Q(1, 1));
209 
210  // See https://eigen.tuxfamily.org/dox/group__TutorialLinearAlgebra.html for
211  // quick overview of possible solver algorithms.
212  // ColPivHouseholderQR is accurate and fast for small systems.
213  Vector3d x{A_vec.colPivHouseholderQr().solve(Q_vec)};
214  Matrix2d X;
215  X << x(0), x(1), x(1), x(2);
216  return X;
217 }

219 MatrixXd SolveReducedRealContinuousLyapunovEquation(
220  const Eigen::Ref<const MatrixXd>& S,
221  const Eigen::Ref<const MatrixXd>& Q_bar) {
222  const int m{static_cast<int>(S.rows())};
223  if (m == 1) return Solve1By1RealContinuousLyapunovEquation(S, Q_bar);
224  if (m == 2) return Solve2By2RealContinuousLyapunovEquation(S, Q_bar);
225  // Notation & partition adapted from SB03MD:
226  //
227  //
228  // S = [s₁₁ s'] Q̅ = [q̅₁₁ q̅'] X̅ = [x̅₁₁ x̅']
229  // [0 S₁] , [q̅ Q₁] , [x̅ X₁]
230  //
231  // where s₁₁ is a 1-by-1 or 2-by-2 matrix.
232  //
233  // s₁₁'x̅₁₁ + x̅₁₁s₁₁ = -q̅₁₁ (3.1)
234  //
235  // S₁'x̅ + x̅s₁₁ = -q̅₁₁ - sx̅₁₁ (3.2)
236  //
237  // S₁'X₁ + X₁S₁ = -Q₁ - sx̅' - x̅s' (3.3)
238 
239  int n{1}; // n equals the size of the n-by-n block in the left-upper
240  // corner of S.
241  // Check if the top block of S is 1-by-1 or 2-by-2.
242  if (!(std::abs(S(1, 0)) < kTolerance * S.norm())) {
243  n = 2;
244  }
245 
246 
247  const MatrixXd s_11{S.topLeftCorner(n, n)};
248  const MatrixXd s{S.transpose().bottomLeftCorner(m - n, n)};
249  const MatrixXd S_1{S.bottomRightCorner(m - n, m - n)};
250  const MatrixXd q_bar_11{Q_bar.topLeftCorner(n, n)};
251  const MatrixXd q_bar{Q_bar.transpose().bottomLeftCorner(m - n, n)};
252  const MatrixXd Q_1{Q_bar.bottomRightCorner(m - n, m - n)};
253 
254  MatrixXd x_bar_11(n, n), x_bar(m - n, n);
255  if (n == 1) {
256  // solving equation 3.1
257  x_bar_11 << Solve1By1RealContinuousLyapunovEquation(s_11, q_bar_11);
258  // solving equation 3.2
259  const MatrixXd lhs{S_1.transpose() +
260  MatrixXd::Identity(S_1.cols(), S_1.rows()) * s_11(0)};
261  const VectorXd rhs{-q_bar - s * x_bar_11(0)};
262  x_bar << lhs.colPivHouseholderQr().solve(rhs);
263  } else {
264  x_bar_11 << Solve2By2RealContinuousLyapunovEquation(s_11, q_bar_11);
265  // solving equation 3.2
266  // The equation reads as S₁'x̅ + x̅s₁₁ = -q̅₁₁ - sx̅₁₁,
267  // where S₁ ∈ ℝᵐ⁻²ˣᵐ⁻²; x̅, q̅, s ∈ ℝᵐ⁻²ˣ² and s₁₁, x̅₁₁ ∈ ℝ²ˣ².
268  // We solve the linear equation by vectorization and feeding it into
269  // colPivHouseHolderQr().
270  //
271  // Notation:
272  //
273  // The elements in s₁₁ are names as the following:
274  // s₁₁ = [s₁₁(0,0) s₁₁(0,1); s₁₁(1,0) s₁₁(1,1)].
275  // Note that eigen starts counting at 0, and not as above at 1.
276  //
277  // The ith column of a matrix is accessed by [i], i.e. the first column
278  // of x̅ is x̅[0].
279  //
280  // Define:
281  //
282  // S_1_vec = [S₁' 0; 0 S₁'] with 0 \in ∈ℝᵐˣᵐ
283  //
284  // S_11_vec = [s₁₁(0,0)*I s₁₁(1,0)*I; s₁₁(0,1)*I s₁₁(1,1)*I] with I ∈ ℝᵐˣᵐ.
285  //
286  // Now define lhs = S_1_vec + S_11_vec.
287  //
288  // x_vec = [x̅[0]' x̅[1]']' where [i] is the ith column
289  //
290  // rhs = - [(q̅+sx̅₁₁)[0]' (q̅+s*x̅₁₁)[1]']'
291  //
292  // Now S₁'x̅ + x̅s₁₁ = -q̅₁₁ - sx̅₁₁ can be rewritten into:
293  //
294  // lhs*x_vec = rhs.
295  //
296  // This expression can be fed into colPivHouseHolderQr() and solved.
297  // Finally, construct x_bar from x_vec.
298  //
299 
300  MatrixXd S_1_vec(2 * (m - 2), 2 * (m - 2));
301  S_1_vec << S_1.transpose(), MatrixXd::Zero(m - 2, m - 2),
302  MatrixXd::Zero(m - 2, m - 2), S_1.transpose();
303 
304  MatrixXd s_11_vec(2 * (m - 2), 2 * (m - 2));
305  s_11_vec << s_11(0, 0) * MatrixXd::Identity(m - 2, m - 2),
306  s_11(1, 0) * MatrixXd::Identity(m - 2, m - 2),
307  s_11(0, 1) * MatrixXd::Identity(m - 2, m - 2),
308  s_11(1, 1) * MatrixXd::Identity(m - 2, m - 2);
309  MatrixXd lhs{S_1_vec + s_11_vec};
310 
311  VectorXd rhs(2 * (m - 2), 1);
312  rhs << (-q_bar - s * x_bar_11).col(0), (-q_bar - s * x_bar_11).col(1);
313 
314  VectorXd x_bar_vec{lhs.colPivHouseholderQr().solve(rhs)};
315  x_bar.col(0) = x_bar_vec.segment(0, m - 2);
316  x_bar.col(1) = x_bar_vec.segment(m - 2, m - 2);
317  }
318 
319  // Set up equation 3.3
320  const Eigen::MatrixXd temp_summand{s * x_bar.transpose()};
321  // Since Q₁ is only an upper triangular matrix, with NAN in the lower part,
322  // Q_NEW_ᵤₚₚₑᵣ is so as well.
323  const Eigen::MatrixXd Q_new_upper{
324  (Q_1 + temp_summand + temp_summand.transpose())};
325  // The solution is found recursively.
326  MatrixXd X_bar(S.cols(), S.rows());
327  X_bar << x_bar_11, x_bar.transpose(), x_bar,
328  SolveReducedRealContinuousLyapunovEquation(S_1, Q_new_upper);
329 
330  return X_bar;
331 }

102 void MocapData::logInitializeLogging()
103 {
104  // Get the current date and time
105  auto now = std::chrono::system_clock::now();
106  std::time_t now_c = std::chrono::system_clock::to_time_t(now);
107 
108  // Get the current date in the desired format
109  std::stringstream date_ss;
110  date_ss << std::put_time(std::localtime(&now_c), "%Y%m%d"); // Current date
111  std::string current_date = date_ss.str();
112 
113  // Create the directory path for the logs
114  std::string log_base_directory = "./src/flightstack/log/" + current_date + "/" + ControlType::getControllerName();
115 
116  // Check if the directory exists, and create it if it doesn't
117  if (!std::filesystem::exists(log_base_directory)) {
118  std::filesystem::create_directories(log_base_directory);
119  }
120 
121  // Check if the "mocap" subdirectory exists within the log_base_directory, create it if needed
122  std::string mocap_directory = log_base_directory + "/mocap";
123  if (!std::filesystem::exists(mocap_directory)) {
124  std::filesystem::create_directories(mocap_directory);
125  }
126 
127  // Generate the mocap log file name with a timestamp
128  std::stringstream mocap_ss;
129  mocap_ss << mocap_directory << "/mocap_log_" << std::put_time(std::localtime(&now_c), "%Y%m%d_%H%M%S") << ".log";
130  std::string mocap_log_filename = mocap_ss.str();
131 
132  MocapData::logger_mocapdata.add_attribute("Tag", attrs::constant< std::string >("MocapTag"));

134  // Define a synchronous sink with a text ostream backend

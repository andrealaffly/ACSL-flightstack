131 {
132  (void)state;
133 
134  // Check if m_udp_driver is not null
135  if (m_udp_driver) {
136  RCLCPP_INFO(get_logger(), "UDP driver is initialized.");
137  }
138 
139  mocap_publisher_ = this->create_publisher<px4_msgs::msg::VehicleOdometry>(
140  "/fmu/in/vehicle_visual_odometry", rclcpp::QoS(100));
141 
142  try {
143 
144  m_udp_driver->init_receiver(m_ip, m_port);
145  m_udp_driver->receiver()->open();
146  m_udp_driver->receiver()->bind();
147  m_udp_driver->receiver()->asyncReceive(
148  std::bind(&UdpReceiverNode::receiver_callback, this, std::placeholders::_1));
149  } catch (const std::exception & ex) {
150  RCLCPP_ERROR(
151  get_logger(), "Error creating UDP receiver: %s:%i - %s",
152  m_ip.c_str(), m_port, ex.what());
153  return LNI::CallbackReturn::FAILURE;
154  }
155 
156  RCLCPP_DEBUG(get_logger(), "UDP receiver successfully configured.");
157 
158  return LNI::CallbackReturn::SUCCESS;

264 // Callback for receiving a UDP datagram
266 void UdpReceiverNode::receiver_callback(const std::vector<uint8_t> & buffer)
267 {
268  // Convert the buffer to a string
269  std::string buffer_str(buffer.begin(), buffer.end());
270 
271  // Check if the message format is valid
272  if (buffer_str.size() < 2 || buffer_str[0] != 'R' || buffer_str[1] != ',') {
273  std::cerr << "Invalid message format!" << std::endl;
274  return; // Exit the function or handle the error accordingly
275  }
276 
277 
278  // Parse the data from the string
279  std::istringstream iss(buffer_str.substr(2)); // Skip "R," prefix
280  char comma;
281  iss >> mc.x >> comma >> mc.y >> comma >> mc.z >> comma // Read in the position
282  >> mc.q0 >> comma >> mc.q1 >> comma >> mc.q2 >> comma >> mc.q3 >> comma // Read in the quaternion
283  >> mc.vx >> comma >> mc.vy >> comma >> mc.vz >> comma // Read in the velocities
284  >> mc.rollspeed >> comma >> mc.pitchspeed >> comma >> mc.yawspeed; // Read in the angular speeds
285 
286  // Debug Mocap parsed data to terminal
287  debugMocapData2screen();
288 
289  // Initilize the Mocap odometry message to publish to the pixhawk
290  px4_msgs::msg::VehicleOdometry out;
291 
292  // \brief: https://github.com/hpaul360/mocap_to_px4/blob/main/mocap_to_px4/converter_member_function.py
293  // Package the message
294  out.timestamp = 0; // Timestamp is automatically set inside PX4
295  out.timestamp_sample = 0; // Timestamp is automatically set inside PX4
296 
297  // Position frame is set to NED in the VICON code.
298  out.position[0] = (float) mc.x;
299  out.position[1] = (float) mc.y;
300  out.position[2] = (float) mc.z;
301  // Orientation frame is set to NED in the VICON code.
302  out.q[0] = (float) mc.q0;
303  out.q[1] = (float) mc.q1;
304  out.q[2] = (float) mc.q2;
305  out.q[3] = (float) mc.q3;
306  out.pose_frame = POSE_FRAME_NED;
307 
308  // Velocity frame is set to NED in the VICON code.
309  out.velocity[0] = (float) mc.vx;
310  out.velocity[1] = (float) mc.vy;
311  out.velocity[2] = (float) mc.vz;
312  out.velocity_frame = VELOCITY_FRAME_NED;
313 
314  // Angular speed is set to 0. We need to pass the velocites to EKF2 in the FRD Body-Fixed
315  // Frame. We do not have the data in the VICON code therefore we set it to 0.
316  out.angular_velocity[0] = 0;
317  out.angular_velocity[1] = 0;
318  out.angular_velocity[2] = 0;
319 
320  // Other Parameters
321  out.position_variance = {0,0,0};
322  out.orientation_variance = {0,0,0};
323  out.velocity_variance = {0,0,0};
324 
325  // Publish the message
326  mocap_publisher_->publish(out);
327 
328 }

283 // Function to initialize the logging system
284 void LogData_MRAC::logInitializeLogging()
285 {
286  // Get the current date and time
287  auto now = std::chrono::system_clock::now();
288  std::time_t now_c = std::chrono::system_clock::to_time_t(now);
289 
290  // Get the current date in the desired format
291  std::stringstream date_ss;
292  date_ss << std::put_time(std::localtime(&now_c), "%Y%m%d"); // Current date
293  std::string current_date = date_ss.str();
294 
295  // Create the directory path for the logs
296  std::string log_base_directory = "./src/flightstack/log/" + current_date + "/" + ControlType::getControllerName();
297 
298  // Check if the directory exists, and create it if it doesn't
299  if (!std::filesystem::exists(log_base_directory)) {
300  std::filesystem::create_directories(log_base_directory);
301  }
302 
303  // Check if the "logs" subdirectory exists within the log_base_directory, create it if needed
304  std::string logs_directory = log_base_directory + "/logs";
305  if (!std::filesystem::exists(logs_directory)) {
306  std::filesystem::create_directories(logs_directory);
307  }
308 
309  // Check if the "gains" subdirectory exists within the log_base_directory, create it if needed
310  std::string gains_directory = log_base_directory + "/gains";
311  if (!std::filesystem::exists(gains_directory)) {
312  std::filesystem::create_directories(gains_directory);
313  }
314 
315  // Generate the log file name with a timestamp
316  std::stringstream log_ss;
317  log_ss << logs_directory << "/log_" << std::put_time(std::localtime(&now_c), "%Y%m%d_%H%M%S") << ".log";
318  std::string log_filename = log_ss.str();
319 
320  // Add the "Tag" attribute with a constant value of "LogDataTag" to the logger
321  LogData_MRAC::logger_logdata.add_attribute("Tag", attrs::constant<std::string>("LogDataTag"));
322 
323  // Define a synchronous sink with a text ostream backend
324  typedef sinks::synchronous_sink<sinks::text_ostream_backend> text_sink;
325  boost::shared_ptr<text_sink> logdata_sink = boost::make_shared<text_sink>();
326 
327  // Add a stream to the backend (in this case, a file stream)
328  logdata_sink->locked_backend()->add_stream(boost::make_shared<std::ofstream>(log_filename));
329 
330  // Set the formatter for the sink
331  logdata_sink->set_formatter(
332  expr::stream
333  << "[" << expr::format_date_time<boost::posix_time::ptime>("TimeStamp", "%Y-%m-%d %H:%M:%S.%f") << "] " // Format date and time
334  << "[" << expr::attr<boost::log::attributes::current_thread_id::value_type>("ThreadID") << "] " // Current thread ID
335  << "[" << expr::attr<std::string>("Tag") << "] " // Tag attribute value
336  << "[" << expr::attr<boost::log::attributes::current_process_id::value_type>("ProcessID") << "] " // Current process ID
337  << "[" << expr::attr<unsigned int>("LineID") << "] " // Line ID
338  << expr::smessage // Log message
339  );
340 
341  // Add the sink to the logging core
342  logging::core::get()->add_sink(logdata_sink);
343 
344  // Set a filter for the sink
345  logdata_sink->set_filter(expr::has_attr("Tag") && expr::attr<std::string>("Tag") == "LogDataTag");
346 
347  logging::add_common_attributes(); // Add attributes like timestamp
348 
349  // Copy the gains file to the gains_directory with a new name containing the time info
350  std::string source_file = "./src/flightstack/params/control/mrac/gains_mrac.json";
351  std::stringstream target_ss;
352  target_ss << gains_directory << "/gains_mrac_" << std::put_time(std::localtime(&now_c), "%Y%m%d_%H%M%S") << ".json";
353  std::string target_file = target_ss.str();
354 
355  if (std::filesystem::exists(source_file)) {
356  std::cout << "GAINS FILE PRESENT" << std::endl;
357  std::filesystem::copy(source_file, target_file);
358  } else {
359  std::cout << "GAINS FILE NOT PRESENT" << std::endl;
360  }
361 
362  logInitializeHeaders();
363 }

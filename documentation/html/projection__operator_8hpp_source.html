126 ConvexFunctionOutput<N> convexFunction(const Eigen::Matrix<double, N, 1>& x,
127  const double x_max, const double epsilon)
128 {
129  // Compute h_function = (x'*x - x_max) / epsilon
130  double h_function = (x.dot(x) - x_max) / epsilon;
131 
132  // Compute dh_dx_jacobian = 2/epsilon * x' (row vector)
133  Eigen::Matrix<double, 1, N> dh_dx_jacobian = (2.0 / epsilon) * x.transpose();
134 
135  // Return a struct containing both h_function and dh_dx_jacobian
136  return ConvexFunctionOutput<N>{h_function, dh_dx_jacobian};

139 template <int N>
149 Eigen::Matrix<double, N, 1> projectionVector(const Eigen::Matrix<double, N, 1>& x,
150  const Eigen::Matrix<double, N, 1>& x_d,
151  const double x_max, const double epsilon)
152 {
153  // Call the convex_function to get h_function and dh_dx_jacobian
154  ConvexFunctionOutput<N> cfo = convexFunction<N>(x, x_max, epsilon);
155  double h_function = cfo.h_function;
156  Eigen::Matrix<double, 1, N> dh_dx_jacobian = cfo.dh_dx_jacobian;
157 
158  // Check if h_function > 0 and dh_dx_jacobian * x_d > 0
159  if (h_function > 0 && dh_dx_jacobian.dot(x_d) > 0)
160  {
161  // Precompute the transpose of dh_dx_jacobian
162  Eigen::Matrix<double, N, 1> dh_dx_jacobian_transpose = dh_dx_jacobian.transpose();
163 
164  // Modify x_d based on the projection formula
165  Eigen::Matrix<double, N, 1> x_d_modified =
166  x_d - h_function * dh_dx_jacobian_transpose * dh_dx_jacobian * x_d /
167  dh_dx_jacobian.dot(dh_dx_jacobian_transpose);
168 
169  return x_d_modified;
170  }
171  else
172  {
173  // If the conditions are not met, return x_d unchanged
174  return x_d;
175  }

178 template <typename Derived1, typename Derived2>
188 auto projectionMatrix(const Eigen::MatrixBase<Derived1>& matrix,
189  const Eigen::MatrixBase<Derived2>& matrix_d,
190  const double x_max, const double epsilon)
191  -> Eigen::Matrix<typename Derived1::Scalar, Derived1::RowsAtCompileTime, Derived1::ColsAtCompileTime>
192 {
193  // Use Eigen::internal::traits to get uniform access to row/col dimensions
194  // Two template types are used because the first input is a Eigen::Map while the
195  // second input is a regular Eigen::Matrix.
196  constexpr int Rows1 = Eigen::internal::traits<Derived1>::RowsAtCompileTime;
197  constexpr int Cols1 = Eigen::internal::traits<Derived1>::ColsAtCompileTime;
198  constexpr int Rows2 = Eigen::internal::traits<Derived2>::RowsAtCompileTime;
199  constexpr int Cols2 = Eigen::internal::traits<Derived2>::ColsAtCompileTime;
200 
201  // Ensure the dimensions of both matrices are the same
202  static_assert(Rows1 == Rows2 && Cols1 == Cols2, "The matrices must have the same dimensions.");
203 
204  // Total number of elements in the matrix
205  constexpr int TotalElements = Rows1 * Cols1;
206 
207  // Reshape the matrix into a column vector (row-major order)
208  Eigen::Map<const Eigen::Matrix<typename Derived1::Scalar, TotalElements, 1>>
209  reshaped_matrix(matrix.derived().data());

211  reshaped_matrix_d(matrix_d.derived().data());

215  projectionVector<TotalElements>(reshaped_matrix, reshaped_matrix_d, x_max, epsilon);
216 
217  // Reshape the projected vector back into a matrix of the same size
218  Eigen::Map<Eigen::Matrix<double, Rows1, Cols1>> projected_matrix(projected_vector.data());

227 template <int N>
240 ConvexFunctionOutput<N> convexFunction(const Eigen::Matrix<double, N, 1>& x,
241  const Eigen::Matrix<double, N, 1>& x_e,
242  const Eigen::Matrix<double, N, N>& S,
243  const double epsilon)
244 {
245  // Compute the difference vector (x - x_e)
246  Eigen::Matrix<double, N, 1> x_diff = x - x_e;
247 
248  // Compute (x - x_e)^T * S * (x - x_e)
249  double quadratic_term = x_diff.transpose() * S * x_diff;
250 
251  // Compute h(x) = ((1 + epsilon) * quadratic_term - 1) / epsilon
252  double h_function = ((1.0 + epsilon) * quadratic_term - 1.0) / epsilon;
253 
254  // Compute dh_dx_jacobian
255  Eigen::Matrix<double, 1, N> dh_dx_jacobian = (2.0 * (1 + epsilon) / epsilon) * x_diff.transpose() * S;
256 
257  // Return a struct containing both h_function and dh_dx_jacobian
258  return ConvexFunctionOutput<N>{h_function, dh_dx_jacobian};
259 }

349 // Function to assign polynomial coefficients to designated variables
350 std::vector<Eigen::MatrixXd> polyCoefAssigning(const std::vector<std::vector<double>>& poly_coeff_matrix_in)
351 {
352  // Get the size of the input matrix
353  int rows = poly_coeff_matrix_in.size();
354  int cols = poly_coeff_matrix_in[0].size();
355 
356  // Calculate the size of the output matrices
357  int out_rows = rows / 3;
358 
359  // Initialize output matrices for x, y, and z variables
360  Eigen::MatrixXd poly_coeff_matrix_out_x(out_rows, cols);
361  Eigen::MatrixXd poly_coeff_matrix_out_y(out_rows, cols);
362  Eigen::MatrixXd poly_coeff_matrix_out_z(out_rows, cols);
363 
364  // Assign polynomial coefficients to designated variables
365  for (int i = 0; i < out_rows; ++i) {
366  for (int j = 0; j < cols; ++j) {
367  poly_coeff_matrix_out_x(i, j) = poly_coeff_matrix_in[3 * i][j];
368  poly_coeff_matrix_out_y(i, j) = poly_coeff_matrix_in[3 * i + 1][j];
369  poly_coeff_matrix_out_z(i, j) = poly_coeff_matrix_in[3 * i + 2][j];
370  }
371  }
372 
373  // Return the output matrices
374  return {poly_coeff_matrix_out_x, poly_coeff_matrix_out_y, poly_coeff_matrix_out_z};
375 }

377 // Function to compute the derivative of the piecewise polynomial coefficient matrix
378 Eigen::MatrixXd polyDerMatrix(const Eigen::MatrixXd& poly_coeff_matrix_in)
379 {
380  // Get the size of the input matrix
381  int rows = poly_coeff_matrix_in.rows();
382  int cols = poly_coeff_matrix_in.cols();
383 
384  // Initialize the output matrix for the derivative
385  Eigen::MatrixXd poly_coeff_matrix_out(rows, cols - 1);
386 
387  // Compute the derivative for each row of the input matrix
388  for (int i = 0; i < rows; ++i) {
389  for (int j = 0; j < cols - 1; ++j) {
390  poly_coeff_matrix_out(i, j) = (cols - 1 - j) * poly_coeff_matrix_in(i, j);
391  }
392  }
393 
394  return poly_coeff_matrix_out;
395 }

397 // Adjusts the time to be fed to the various polynomials and identifies the segment of the trajectory

461 // Compute the user-defined yaw_dot from the trajectory velocity in the XY plane
462 double yawDotComputation(const Eigen::VectorXd& Vx_coef, const Eigen::VectorXd& Vy_coef,
463  const Eigen::VectorXd& Ax_coef, const Eigen::VectorXd& Ay_coef, double t)
464 {
465  // Compute the acceleration components Ax and Ay at time t using polynomial coefficients
466  double Ax = evaluatePolynomial(Ax_coef, t);
467  double Ay = evaluatePolynomial(Ay_coef, t);
468 
469  // Compute the acceleration angle in the complex plane
470  double acceleration_angle = std::atan2(Ay, Ax);
471 
472  // Compute the velocity and acceleration norms
473  double velocity_norm = norm2D(Vx_coef, Vy_coef, t);
474  double acceleration_norm = norm2D(Ax_coef, Ay_coef, t);
475 
476  // Compute the yaw angle
477  double yaw = yawComputation(Vx_coef, Vy_coef, t);
478 
479  // Compute the yaw rate
480  double yaw_dot = (acceleration_norm / velocity_norm) * std::sin(acceleration_angle - yaw);
481 
482  return yaw_dot;
483 }

485 // Compute the user-defined yaw_dot_dot from the trajectory velocity in the XY plane
486 double yawDotDotComputation(const Eigen::VectorXd& Vx_coef, const Eigen::VectorXd& Vy_coef,
487  const Eigen::VectorXd& Ax_coef, const Eigen::VectorXd& Ay_coef,
488  const Eigen::VectorXd& Jx_coef, const Eigen::VectorXd& Jy_coef,
489  double t)
490 {
491  // Compute the jerk components Jx and Jy at time t using polynomial coefficients
492  double Jx = evaluatePolynomial(Jx_coef, t);
493  double Jy = evaluatePolynomial(Jy_coef, t);
494 
495  // Compute the jerk angle in the complex plane
496  double jerk_angle = std::atan2(Jy, Jx);
497 
498  // Compute the velocity and jerk norms
499  double velocity_norm = norm2D(Vx_coef, Vy_coef, t);
500  double jerk_norm = norm2D(Jx_coef, Jy_coef, t);
501 
502  // Compute the derivative of the velocity norm
503  double velocity_norm_prime = norm2Dderivative(Vx_coef, Vy_coef, Ax_coef, Ay_coef, t);
504 
505  // Compute the yaw angle and its derivative
506  double yaw = yawComputation(Vx_coef, Vy_coef, t);
507  double yaw_dot = yawDotComputation(Vx_coef, Vy_coef, Ax_coef, Ay_coef, t);
508 
509  // Compute the second derivative of yaw angle
510  double yaw_dot_dot = (jerk_norm * std::sin(jerk_angle - yaw) -
511  2 * velocity_norm_prime * yaw_dot) / velocity_norm;
512 
513  return yaw_dot_dot;
514 }

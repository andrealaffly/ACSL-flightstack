106 /*
107  Convex function to compute h and dh_dx.
108  The inner convex set is a ball of radius sqrt(x_max)
109  The outer convex set is a ball of radius sqrt(x_max + epsilon)
110 */
111 template <int N>
112 ConvexFunctionOutput<N> convexFunction(const Eigen::Matrix<double, N, 1>& x,
113  const double x_max, const double epsilon)
114 {
115  // Compute h_function = (x'*x - x_max) / epsilon
116  double h_function = (x.dot(x) - x_max) / epsilon;
117 
118  // Compute dh_dx_jacobian = 2/epsilon * x' (row vector)
119  Eigen::Matrix<double, 1, N> dh_dx_jacobian = (2.0 / epsilon) * x.transpose();
120 
121  // Return a struct containing both h_function and dh_dx_jacobian
122  return ConvexFunctionOutput<N>{h_function, dh_dx_jacobian};
123 }

125 /*
126  Function to project the vector x_d based on the projection_operator::ball::convexFunction()
127  x_d: derivative wrt time of x (RHS of its dynamics equation)
128 */
129 template <int N>
130 Eigen::Matrix<double, N, 1> projectionVector(const Eigen::Matrix<double, N, 1>& x,
131  const Eigen::Matrix<double, N, 1>& x_d,
132  const double x_max, const double epsilon)
133 {
134  // Call the convex_function to get h_function and dh_dx_jacobian
135  ConvexFunctionOutput<N> cfo = convexFunction<N>(x, x_max, epsilon);
136  double h_function = cfo.h_function;
137  Eigen::Matrix<double, 1, N> dh_dx_jacobian = cfo.dh_dx_jacobian;
138 
139  // Check if h_function > 0 and dh_dx_jacobian * x_d > 0
140  if (h_function > 0 && dh_dx_jacobian.dot(x_d) > 0)
141  {
142  // Precompute the transpose of dh_dx_jacobian
143  Eigen::Matrix<double, N, 1> dh_dx_jacobian_transpose = dh_dx_jacobian.transpose();
144 
145  // Modify x_d based on the projection formula
146  Eigen::Matrix<double, N, 1> x_d_modified =
147  x_d - h_function * dh_dx_jacobian_transpose * dh_dx_jacobian * x_d /
148  dh_dx_jacobian.dot(dh_dx_jacobian_transpose);
149 
150  return x_d_modified;
151  }
152  else
153  {
154  // If the conditions are not met, return x_d unchanged
155  return x_d;
156  }
157 }

202 /*
203  Convex function to compute h and dh_dx.
204  The outer convex set is an ellipsoid centered in x_e with semi-axis length described by the diagonal elements
205  that populate the S matrix. If S = diag(1/a^2, 1/b^2, 1/c^2) then the semi-axis lengths are [a, b, c].
206  The inner convex set is the outer convex set ellipsoid scaled by the coefficient alpha = 1/sqrt(1 + epsilon).
207 */
208 template <int N>
209 ConvexFunctionOutput<N> convexFunction(const Eigen::Matrix<double, N, 1>& x,
210  const Eigen::Matrix<double, N, 1>& x_e,
211  const Eigen::Matrix<double, N, N>& S,
212  const double epsilon)
213 {
214  // Compute the difference vector (x - x_e)
215  Eigen::Matrix<double, N, 1> x_diff = x - x_e;
216 
217  // Compute (x - x_e)^T * S * (x - x_e)
218  double quadratic_term = x_diff.transpose() * S * x_diff;
219 
220  // Compute h(x) = ((1 + epsilon) * quadratic_term - 1) / epsilon
221  double h_function = ((1.0 + epsilon) * quadratic_term - 1.0) / epsilon;
222 
223  // Compute dh_dx_jacobian
224  Eigen::Matrix<double, 1, N> dh_dx_jacobian = (2.0 * (1 + epsilon) / epsilon) * x_diff.transpose() * S;
225 
226  // Return a struct containing both h_function and dh_dx_jacobian
227  return ConvexFunctionOutput<N>{h_function, dh_dx_jacobian};
228 }

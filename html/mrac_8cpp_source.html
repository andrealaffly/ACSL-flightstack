222 void MRAC::initializeControllerParameters(VehicleInfo& vehicle_info, GainsMRAC& gains_)
223 {
224  // Initialize to zero the 6x6 matrix and set the top-right 3x3 block as follows
225  gains_.A_translational.setZero();
226  gains_.A_translational.block<3, 3>(0, 3) = Eigen::Matrix3d::Identity();
227

229  gains_.B_translational.setZero();
230  gains_.B_translational.block<3, 3>(3, 0) = Eigen::Matrix3d::Identity();
231 
232  // Initialize to zero the 6x6 matrix
233  gains_.A_ref_translational.setZero();
234  // Set the top-right 3x3 block as follows
235  gains_.A_ref_translational.block<3, 3>(0, 3) = Eigen::Matrix3d::Identity();
236  // Set the bottom-left 3x3 block as follows
237  gains_.A_ref_translational.block<3, 3>(3, 0) = -gains_.KP_refmod_translational;
238  // Set the bottom-right 3x3 block as follows
239  gains_.A_ref_translational.block<3, 3>(3, 3) = -gains_.KD_refmod_translational;
240 
241  // Initialize to zero the 6x3 matrix and set the bottom 3x3 block as follows
242  gains_.B_ref_translational.setZero();
243  gains_.B_ref_translational.block<3, 3>(3, 0) = (1 / vehicle_info.mass) * Eigen::Matrix3d::Identity();
244 
245  // Initialize to zero the 3x3 matrix
246  gains_.A_rotational.setZero();
247 
248  // Set the 3x3 matrix as follows
249  gains_.B_rotational = Eigen::Matrix3d::Identity();
250 
251  // Set the 3x3 matrix as follows
252  gains_.A_ref_rotational = -gains_.KP_omega_ref_rotational;
253 
254  // Set the 3x3 matrix as follows
255  gains_.B_ref_rotational = Eigen::Matrix3d::Identity();
256 
257  // Solve the continuos Lyapunov equation to compute P_translational
258  gains_.P_translational = RealContinuousLyapunovEquation(gains_.A_ref_translational,
259  gains_.Q_translational);
260 
261  // Solve the continuos Lyapunov equation to compute P_rotational

263  gains_.Q_rotational);
264 
265  // Projection operator. Generate the S matrices from their ellipsoid semi-axis terms contained in S_diagonal

267  gains_.S_r_translational = projection_operator::generateEllipsoidMatrixFromDiagonal(gains_.S_diagonal_r_translational);

353 /*
354  OUTER LOOP CONTROLLER
355 */
356 void MRAC::computeOuterLoop(ControlInternalMembers& cim,
357  VehicleInfo& vehicle_info,
358  StateController& state_,
359  ControlReferences& cr,
360  GainsMRAC& gains_,
361  ControllerSpecificInternalMembers& csim_)
362 {
363  // Compute the translational_position_error [actual - reference_model]
364  this->computeTranslationalPositionError(cim, cr.position, state_.x_ref_translational.head<3>());
365  // Compute the translational_position_error_ref [reference_model - user_defined_trajectory]
366  csim_.translational_position_error_ref = state_.x_ref_translational.head<3>() - cr.user_defined_position;
367 
368  // Reference command input [reference_model - user_defined_trajectory]
369  csim_.r_cmd_translational = vehicle_info.mass * (
370  - gains_.KI_refmod_translational * state_.integral_translational_position_error_ref
371  + gains_.KP_refmod_translational * cr.user_defined_position
372  + gains_.KD_refmod_translational * cr.user_defined_velocity
373  + cr.user_defined_acceleration
374  );
375 
376  // Reference model
377  csim_.x_ref_dot_translational = gains_.A_ref_translational * state_.x_ref_translational
378  + gains_.B_ref_translational * csim_.r_cmd_translational;
379 
380  // Baseline PID controller terms [actual - reference_model]
381  csim_.outer_loop_dynamic_inversion = - vehicle_info.mass * GRAVITATIONAL_ACCELERATION * this->e3_basis;
382  csim_.outer_loop_P = - gains_.KP_translational * cim.translational_position_error;
383  csim_.outer_loop_D = - gains_.KD_translational * (cr.velocity - state_.x_ref_translational.tail<3>());
384  csim_.outer_loop_I = - gains_.KI_translational * state_.integral_translational_position_error;
385  // Baseline PID controller [actual - reference_model]
386  csim_.mu_PID_baseline_translational = csim_.outer_loop_dynamic_inversion +
387  vehicle_info.mass * (
388  csim_.outer_loop_P
389  + csim_.outer_loop_D
390  + csim_.outer_loop_I
391  + csim_.x_ref_dot_translational.tail<3>()
392  );
393 
394  // Velocity expressed in body/local reference frame
395  Eigen::Vector3d velocity_expr_in_local = cim.rotation_matrix_321_global_to_local * cr.velocity;
396  // Regressor vector
397  Eigen::Vector3d regressor_vector_translational = -0.5 * velocity_expr_in_local.norm() * velocity_expr_in_local;
398 
399  // Augmented regressor vector [includes the baseline controller]
400  csim_.augmented_regressor_vector_translational.head<3>() = csim_.mu_PID_baseline_translational;
401  csim_.augmented_regressor_vector_translational.tail<3>() = regressor_vector_translational;
402 
403  // Translational dynamics state
404  Eigen::Matrix<double, 6, 1> x_translational;
405  x_translational.head<3>() = cr.position;
406  x_translational.tail<3>() = cr.velocity;
407 
408  // Tracking error [transposed]
409  Eigen::Matrix<double, 1, 6> e_translational_transposed = (x_translational - state_.x_ref_translational).transpose();
410 
411  // Precomputing (e^T * P * B)
412  Eigen::Matrix<double, 1, 3> eTranspose_P_B_translational = e_translational_transposed *
413  gains_.P_translational *
414  gains_.B_translational;
415 
416  // Precomputing the norm of (e^T * P * B)
417  double eTranspose_P_B_norm_translational = eTranspose_P_B_translational.norm();
418 
419  // Computing the scalar value output from the dead-zone modification modulation function
420  csim_.dead_zone_value_translational = deadZoneModulationFunction(e_translational_transposed,
421  gains_.dead_zone_delta_translational,
422  gains_.dead_zone_e0_translational);
423 
424  // Adaptive law K_hat_x

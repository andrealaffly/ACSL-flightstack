163 // Function to initialize the logging system
164 void LogData_PID::logInitializeLogging()
165 {
166  // Get the current date and time
167  auto now = std::chrono::system_clock::now();
168  std::time_t now_c = std::chrono::system_clock::to_time_t(now);
169 
170  // Get the current date in the desired format
171  std::stringstream date_ss;
172  date_ss << std::put_time(std::localtime(&now_c), "%Y%m%d"); // Current date
173  std::string current_date = date_ss.str();
174 
175  // Create the directory path for the logs
176  std::string log_base_directory = "./src/flightstack/log/" + current_date + "/" + ControlType::getControllerName();
177 
178  // Check if the directory exists, and create it if it doesn't
179  if (!std::filesystem::exists(log_base_directory)) {
180  std::filesystem::create_directories(log_base_directory);
181  }
182 
183  // Check if the "logs" subdirectory exists within the log_base_directory, create it if needed
184  std::string logs_directory = log_base_directory + "/logs";
185  if (!std::filesystem::exists(logs_directory)) {
186  std::filesystem::create_directories(logs_directory);
187  }
188 
189  // Check if the "gains" subdirectory exists within the log_base_directory, create it if needed
190  std::string gains_directory = log_base_directory + "/gains";
191  if (!std::filesystem::exists(gains_directory)) {
192  std::filesystem::create_directories(gains_directory);
193  }
194 
195  // Generate the log file name with a timestamp
196  std::stringstream log_ss;
197  log_ss << logs_directory << "/log_" << std::put_time(std::localtime(&now_c), "%Y%m%d_%H%M%S") << ".log";
198  std::string log_filename = log_ss.str();
199 
200  // Add the "Tag" attribute with a constant value of "LogDataTag" to the logger
201  LogData_PID::logger_logdata.add_attribute("Tag", attrs::constant<std::string>("LogDataTag"));
202 
203  // Define a synchronous sink with a text ostream backend
204  typedef sinks::synchronous_sink<sinks::text_ostream_backend> text_sink;
205  boost::shared_ptr<text_sink> logdata_sink = boost::make_shared<text_sink>();
206 
207  // Add a stream to the backend (in this case, a file stream)
208  logdata_sink->locked_backend()->add_stream(boost::make_shared<std::ofstream>(log_filename));
209 
210  // Set the formatter for the sink
211  logdata_sink->set_formatter(
212  expr::stream
213  << "[" << expr::format_date_time<boost::posix_time::ptime>("TimeStamp", "%Y-%m-%d %H:%M:%S.%f") << "] " // Format date and time
214  << "[" << expr::attr<boost::log::attributes::current_thread_id::value_type>("ThreadID") << "] " // Current thread ID
215  << "[" << expr::attr<std::string>("Tag") << "] " // Tag attribute value
216  << "[" << expr::attr<boost::log::attributes::current_process_id::value_type>("ProcessID") << "] " // Current process ID
217  << "[" << expr::attr<unsigned int>("LineID") << "] " // Line ID
218  << expr::smessage // Log message
219  );
220 
221  // Add the sink to the logging core
222  logging::core::get()->add_sink(logdata_sink);
223 
224  // Set a filter for the sink
225  logdata_sink->set_filter(expr::has_attr("Tag") && expr::attr<std::string>("Tag") == "LogDataTag");
226 
227  logging::add_common_attributes(); // Add attributes like timestamp
228 
229  // Copy the gains file to the gains_directory with a new name containing the time info
230  std::string source_file = "./src/flightstack/params/control/pid/gains_pid.json";
231  std::stringstream target_ss;
232  target_ss << gains_directory << "/gains_pid_" << std::put_time(std::localtime(&now_c), "%Y%m%d_%H%M%S") << ".json";
233  std::string target_file = target_ss.str();
234 
235  if (std::filesystem::exists(source_file)) {
236  std::cout << "GAINS FILE PRESENT" << std::endl;
237  std::filesystem::copy(source_file, target_file);
238  } else {
239  std::cout << "GAINS FILE NOT PRESENT" << std::endl;
240  }
241 
242  logInitializeHeaders();
243 }

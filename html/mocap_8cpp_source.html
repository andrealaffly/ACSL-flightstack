147 LNI::CallbackReturn UdpReceiverNode::on_configure(const lc::State & state)
148 {
149  (void)state;
150 
151  // Check if m_udp_driver is not null
152  if (m_udp_driver) {
153  RCLCPP_INFO(get_logger(), "UDP driver is initialized.");
154  }
155 
156  mocap_publisher_ = this->create_publisher<px4_msgs::msg::VehicleOdometry>(
157  "/fmu/in/vehicle_visual_odometry", rclcpp::QoS(100));
158 
159  try {
160 
161  m_udp_driver->init_receiver(m_ip, m_port);
162  m_udp_driver->receiver()->open();
163  m_udp_driver->receiver()->bind();
164  m_udp_driver->receiver()->asyncReceive(
165  std::bind(&UdpReceiverNode::receiver_callback, this, std::placeholders::_1));
166  } catch (const std::exception & ex) {
167  RCLCPP_ERROR(
168  get_logger(), "Error creating UDP receiver: %s:%i - %s",
169  m_ip.c_str(), m_port, ex.what());
170  return LNI::CallbackReturn::FAILURE;
171  }
172 
173  RCLCPP_DEBUG(get_logger(), "UDP receiver successfully configured.");
174 
175  return LNI::CallbackReturn::SUCCESS;
176 }

299 void UdpReceiverNode::receiver_callback(const std::vector<uint8_t> & buffer)
300 {
301  // Convert the buffer to a string
302  std::string buffer_str(buffer.begin(), buffer.end());
303 
304  // Check if the message format is valid
305  if (buffer_str.size() < 2 || buffer_str[0] != 'R' || buffer_str[1] != ',') {
306  std::cerr << "Invalid message format!" << std::endl;
307  return; // Exit the function or handle the error accordingly
308  }
309 
310 
311  // Parse the data from the string
312  std::istringstream iss(buffer_str.substr(2)); // Skip "R," prefix
313  char comma;
314  iss >> mc.x >> comma >> mc.y >> comma >> mc.z >> comma // Read in the position
315  >> mc.q0 >> comma >> mc.q1 >> comma >> mc.q2 >> comma >> mc.q3 >> comma // Read in the quaternion
316  >> mc.vx >> comma >> mc.vy >> comma >> mc.vz >> comma // Read in the velocities
317  >> mc.rollspeed >> comma >> mc.pitchspeed >> comma >> mc.yawspeed; // Read in the angular speeds
318 
319  // Debug Mocap parsed data to terminal
320  debugMocapData2screen();
321 
322  // Initilize the Mocap odometry message to publish to the pixhawk
323  px4_msgs::msg::VehicleOdometry out;
324 
325  // \brief: https://github.com/hpaul360/mocap_to_px4/blob/main/mocap_to_px4/converter_member_function.py
326  // Package the message
327  out.timestamp = 0; // Timestamp is automatically set inside PX4
328  out.timestamp_sample = 0; // Timestamp is automatically set inside PX4
329 
330  // Position frame is set to NED in the VICON code.
331  out.position[0] = (float) mc.x;
332  out.position[1] = (float) mc.y;
333  out.position[2] = (float) mc.z;
334  // Orientation frame is set to NED in the VICON code.
335  out.q[0] = (float) mc.q0;
336  out.q[1] = (float) mc.q1;
337  out.q[2] = (float) mc.q2;
338  out.q[3] = (float) mc.q3;
339  out.pose_frame = POSE_FRAME_NED;
340 
341  // Velocity frame is set to NED in the VICON code.
342  out.velocity[0] = (float) mc.vx;
343  out.velocity[1] = (float) mc.vy;
344  out.velocity[2] = (float) mc.vz;
345  out.velocity_frame = VELOCITY_FRAME_NED;
346 
347  // Angular speed is set to 0. We need to pass the velocites to EKF2 in the FRD Body-Fixed
348  // Frame. We do not have the data in the VICON code therefore we set it to 0.
349  out.angular_velocity[0] = 0;
350  out.angular_velocity[1] = 0;
351  out.angular_velocity[2] = 0;
352 
353  // Other Parameters
354  out.position_variance = {0,0,0};
355  out.orientation_variance = {0,0,0};
356  out.velocity_variance = {0,0,0};
357 
358  // Publish the message
359  mocap_publisher_->publish(out);
360 
361 }

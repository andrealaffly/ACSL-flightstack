105 MatrixXd RealContinuousLyapunovEquation(const Eigen::Ref<const MatrixXd>& A,
106  const Eigen::Ref<const MatrixXd>& Q) {
107  if (A.rows() != A.cols() || Q.rows() != Q.cols() || A.rows() != Q.rows()) {
108  throw std::runtime_error(
109  "RealContinuousLyapunovEquation(): A and Q must be square and of equal "
110  "size!");
111  }
112 
113  if (!is_approx_equal_abstol(Q, Q.transpose(), 1e-10)) {
114  throw std::runtime_error("Q is not symmetric.");
115  }
116 
117  // The cases where A is 1-by-1 or 2-by-2 are caught as we have an efficient
118  // methods at hand to solve these without the use of Schur factorization.
119  if (static_cast<int>(A.cols()) == 1) {
120  if (is_zero(A(0, 0))) {
121  throw std::runtime_error(
122  "RealContinuousLyapunovEquation(): Solution is not unique!");
123  }
124  return internal::Solve1By1RealContinuousLyapunovEquation(A, Q);
125  }
126  if (static_cast<int>(A.cols()) == 2) {
127  // If A has two real eigenvalues λ₁ and λ₂, then we check (1) if λ₁ or λ₂ is
128  // 0, i.e. if the determinant of A is 0; (2) if λ₁ + λ₂ = 0, i.e., if the
129  // trace of A is 0. If A has two complex eigenvalues λ₁ and λ̅₁, then we
130  // check if λ₁ + λ̅₁ = 0, i.e., if the trace of A is 0.
131  if (is_zero(A(0, 0) + A(1, 1)) ||
132  is_zero(A(0, 0) * A(1, 1) - A(1, 0) * A(0, 1))) {
133  throw std::runtime_error(
134  "RealContinuousLyapunovEquation(): Solution is not unique!");
135  }
136  // Reducing Q to upper triangular form.
137  MatrixXd Q_upper{Q};
138  Q_upper(1, 0) = NAN;
139  return internal::Solve2By2RealContinuousLyapunovEquation(A, Q_upper);
140  }
141  VectorXcd eig_val{A.eigenvalues()};
142  for (int i = 0; i < eig_val.size(); ++i) {
143  for (int j = i; j < eig_val.size(); ++j) {
144  std::complex<double> eig_sum = eig_val(i) + std::conj(eig_val(j));
145  if (is_zero(eig_sum.real()) && is_zero(eig_sum.imag())) {
146  throw std::runtime_error(
147  "RealContinuousLyapunovEquation(): Solution is not unique!");
148  }
149  }
150  }
151  // Transform into reduced form S'*X̅ + X̅ *S = -Q̅, where A = U*S*U', X̅ =
152  // U'*X*U, Q̅ = U'*Q*U, U is the unitary matrix, and S the reduced form. Note
153  // that the expression for X̅ in [2] is incorrect.
154  Eigen::RealSchur<MatrixXd> schur(A);
155  const MatrixXd U{schur.matrixU()};
156  const MatrixXd S{schur.matrixT()};
157  if (schur.info() != Eigen::Success) {
158  throw std::runtime_error(
159  "RealContinuousLyapunovEquation(): Schur factorization failed.");
160  }
161  // Reduce the symmetric Q̅ to its upper triangular form Q̅_ᵤₚₚₑᵣ.
162  MatrixXd Q_bar_upper{MatrixXd::Constant(Q.rows(), Q.cols(), NAN)};
163  Q_bar_upper.triangularView<Eigen::Upper>() = U.transpose() * Q * U;
164  return (U *
165  internal::SolveReducedRealContinuousLyapunovEquation(S, Q_bar_upper) *
166  U.transpose());
167 }

180 Matrix2d Solve2By2RealContinuousLyapunovEquation(
181  const Eigen::Ref<const Matrix2d>& A, const Eigen::Ref<const Matrix2d>& Q) {
182  // Rewrite AᵀX + XA + Q = 0 (case where A,Q are 2-by-2) into linear set A_vec
183  // *vec(X) = -vec(Q) and solve for X.
184  // Only the upper triangular part of Q is used, thus it is treated to be
185  // symmetric.
186  Matrix3d A_vec;
187  A_vec << 2 * A(0, 0), 2 * A(1, 0), 0, A(0, 1), A(0, 0) + A(1, 1), A(1, 0), 0,
188  2 * A(0, 1), 2 * A(1, 1);
189  const Vector3d Q_vec(-Q(0, 0), -Q(0, 1), -Q(1, 1));
190 
191  // See https://eigen.tuxfamily.org/dox/group__TutorialLinearAlgebra.html for
192  // quick overview of possible solver algorithms.
193  // ColPivHouseholderQR is accurate and fast for small systems.
194  Vector3d x{A_vec.colPivHouseholderQr().solve(Q_vec)};
195  Matrix2d X;
196  X << x(0), x(1), x(1), x(2);
197  return X;
198 }

200 MatrixXd SolveReducedRealContinuousLyapunovEquation(
201  const Eigen::Ref<const MatrixXd>& S,
202  const Eigen::Ref<const MatrixXd>& Q_bar) {
203  const int m{static_cast<int>(S.rows())};
204  if (m == 1) return Solve1By1RealContinuousLyapunovEquation(S, Q_bar);
205  if (m == 2) return Solve2By2RealContinuousLyapunovEquation(S, Q_bar);
206  // Notation & partition adapted from SB03MD:
207  //
208  //
209  // S = [s₁₁ s'] Q̅ = [q̅₁₁ q̅'] X̅ = [x̅₁₁ x̅']
210  // [0 S₁] , [q̅ Q₁] , [x̅ X₁]
211  //
212  // where s₁₁ is a 1-by-1 or 2-by-2 matrix.
213  //
214  // s₁₁'x̅₁₁ + x̅₁₁s₁₁ = -q̅₁₁ (3.1)
215  //
216  // S₁'x̅ + x̅s₁₁ = -q̅₁₁ - sx̅₁₁ (3.2)
217  //
218  // S₁'X₁ + X₁S₁ = -Q₁ - sx̅' - x̅s' (3.3)
219 
220  int n{1}; // n equals the size of the n-by-n block in the left-upper
221  // corner of S.
222  // Check if the top block of S is 1-by-1 or 2-by-2.
223  if (!(std::abs(S(1, 0)) < kTolerance * S.norm())) {
224  n = 2;
225  }
226 
227 
228  const MatrixXd s_11{S.topLeftCorner(n, n)};
229  const MatrixXd s{S.transpose().bottomLeftCorner(m - n, n)};
230  const MatrixXd S_1{S.bottomRightCorner(m - n, m - n)};
231  const MatrixXd q_bar_11{Q_bar.topLeftCorner(n, n)};
232  const MatrixXd q_bar{Q_bar.transpose().bottomLeftCorner(m - n, n)};
233  const MatrixXd Q_1{Q_bar.bottomRightCorner(m - n, m - n)};
234 
235  MatrixXd x_bar_11(n, n), x_bar(m - n, n);
236  if (n == 1) {
237  // solving equation 3.1
238  x_bar_11 << Solve1By1RealContinuousLyapunovEquation(s_11, q_bar_11);
239  // solving equation 3.2
240  const MatrixXd lhs{S_1.transpose() +
241  MatrixXd::Identity(S_1.cols(), S_1.rows()) * s_11(0)};
242  const VectorXd rhs{-q_bar - s * x_bar_11(0)};
243  x_bar << lhs.colPivHouseholderQr().solve(rhs);
244  } else {
245  x_bar_11 << Solve2By2RealContinuousLyapunovEquation(s_11, q_bar_11);
246  // solving equation 3.2
247  // The equation reads as S₁'x̅ + x̅s₁₁ = -q̅₁₁ - sx̅₁₁,
248  // where S₁ ∈ ℝᵐ⁻²ˣᵐ⁻²; x̅, q̅, s ∈ ℝᵐ⁻²ˣ² and s₁₁, x̅₁₁ ∈ ℝ²ˣ².
249  // We solve the linear equation by vectorization and feeding it into
250  // colPivHouseHolderQr().
251  //
252  // Notation:
253  //
254  // The elements in s₁₁ are names as the following:
255  // s₁₁ = [s₁₁(0,0) s₁₁(0,1); s₁₁(1,0) s₁₁(1,1)].
256  // Note that eigen starts counting at 0, and not as above at 1.
257  //
258  // The ith column of a matrix is accessed by [i], i.e. the first column
259  // of x̅ is x̅[0].
260  //
261  // Define:
262  //
263  // S_1_vec = [S₁' 0; 0 S₁'] with 0 \in ∈ℝᵐˣᵐ
264  //
265  // S_11_vec = [s₁₁(0,0)*I s₁₁(1,0)*I; s₁₁(0,1)*I s₁₁(1,1)*I] with I ∈ ℝᵐˣᵐ.
266  //
267  // Now define lhs = S_1_vec + S_11_vec.
268  //
269  // x_vec = [x̅[0]' x̅[1]']' where [i] is the ith column
270  //
271  // rhs = - [(q̅+sx̅₁₁)[0]' (q̅+s*x̅₁₁)[1]']'
272  //
273  // Now S₁'x̅ + x̅s₁₁ = -q̅₁₁ - sx̅₁₁ can be rewritten into:
274  //
275  // lhs*x_vec = rhs.
276  //
277  // This expression can be fed into colPivHouseHolderQr() and solved.
278  // Finally, construct x_bar from x_vec.
279  //
280 
281  MatrixXd S_1_vec(2 * (m - 2), 2 * (m - 2));
282  S_1_vec << S_1.transpose(), MatrixXd::Zero(m - 2, m - 2),
283  MatrixXd::Zero(m - 2, m - 2), S_1.transpose();
284 
285  MatrixXd s_11_vec(2 * (m - 2), 2 * (m - 2));
286  s_11_vec << s_11(0, 0) * MatrixXd::Identity(m - 2, m - 2),
287  s_11(1, 0) * MatrixXd::Identity(m - 2, m - 2),
288  s_11(0, 1) * MatrixXd::Identity(m - 2, m - 2),
289  s_11(1, 1) * MatrixXd::Identity(m - 2, m - 2);
290  MatrixXd lhs{S_1_vec + s_11_vec};
291 
292  VectorXd rhs(2 * (m - 2), 1);
293  rhs << (-q_bar - s * x_bar_11).col(0), (-q_bar - s * x_bar_11).col(1);
294 
295  VectorXd x_bar_vec{lhs.colPivHouseholderQr().solve(rhs)};
296  x_bar.col(0) = x_bar_vec.segment(0, m - 2);
297  x_bar.col(1) = x_bar_vec.segment(m - 2, m - 2);
298  }
299 
300  // Set up equation 3.3
301  const Eigen::MatrixXd temp_summand{s * x_bar.transpose()};
302  // Since Q₁ is only an upper triangular matrix, with NAN in the lower part,
303  // Q_NEW_ᵤₚₚₑᵣ is so as well.
304  const Eigen::MatrixXd Q_new_upper{
305  (Q_1 + temp_summand + temp_summand.transpose())};
306  // The solution is found recursively.
307  MatrixXd X_bar(S.cols(), S.rows());
308  X_bar << x_bar_11, x_bar.transpose(), x_bar,
309  SolveReducedRealContinuousLyapunovEquation(S_1, Q_new_upper);
310 
311  return X_bar;
312 }

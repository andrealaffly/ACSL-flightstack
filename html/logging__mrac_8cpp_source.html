276 // Function to initialize the logging system
277 void LogData_MRAC::logInitializeLogging()
278 {
279  // Get the current date and time
280  auto now = std::chrono::system_clock::now();
281  std::time_t now_c = std::chrono::system_clock::to_time_t(now);
282 
283  // Get the current date in the desired format
284  std::stringstream date_ss;
285  date_ss << std::put_time(std::localtime(&now_c), "%Y%m%d"); // Current date
286  std::string current_date = date_ss.str();
287 
288  // Create the directory path for the logs
289  std::string log_base_directory = "./src/flightstack/log/" + current_date + "/" + ControlType::getControllerName();
290 
291  // Check if the directory exists, and create it if it doesn't
292  if (!std::filesystem::exists(log_base_directory)) {
293  std::filesystem::create_directories(log_base_directory);
294  }
295 
296  // Check if the "logs" subdirectory exists within the log_base_directory, create it if needed
297  std::string logs_directory = log_base_directory + "/logs";
298  if (!std::filesystem::exists(logs_directory)) {
299  std::filesystem::create_directories(logs_directory);
300  }
301 
302  // Check if the "gains" subdirectory exists within the log_base_directory, create it if needed
303  std::string gains_directory = log_base_directory + "/gains";
304  if (!std::filesystem::exists(gains_directory)) {
305  std::filesystem::create_directories(gains_directory);
306  }
307 
308  // Generate the log file name with a timestamp
309  std::stringstream log_ss;
310  log_ss << logs_directory << "/log_" << std::put_time(std::localtime(&now_c), "%Y%m%d_%H%M%S") << ".log";
311  std::string log_filename = log_ss.str();
312 
313  // Add the "Tag" attribute with a constant value of "LogDataTag" to the logger
314  LogData_MRAC::logger_logdata.add_attribute("Tag", attrs::constant<std::string>("LogDataTag"));
315 
316  // Define a synchronous sink with a text ostream backend
317  typedef sinks::synchronous_sink<sinks::text_ostream_backend> text_sink;
318  boost::shared_ptr<text_sink> logdata_sink = boost::make_shared<text_sink>();
319 
320  // Add a stream to the backend (in this case, a file stream)
321  logdata_sink->locked_backend()->add_stream(boost::make_shared<std::ofstream>(log_filename));
322 
323  // Set the formatter for the sink
324  logdata_sink->set_formatter(
325  expr::stream
326  << "[" << expr::format_date_time<boost::posix_time::ptime>("TimeStamp", "%Y-%m-%d %H:%M:%S.%f") << "] " // Format date and time
327  << "[" << expr::attr<boost::log::attributes::current_thread_id::value_type>("ThreadID") << "] " // Current thread ID
328  << "[" << expr::attr<std::string>("Tag") << "] " // Tag attribute value
329  << "[" << expr::attr<boost::log::attributes::current_process_id::value_type>("ProcessID") << "] " // Current process ID
330  << "[" << expr::attr<unsigned int>("LineID") << "] " // Line ID
331  << expr::smessage // Log message
332  );
333 
334  // Add the sink to the logging core
335  logging::core::get()->add_sink(logdata_sink);
336 
337  // Set a filter for the sink
338  logdata_sink->set_filter(expr::has_attr("Tag") && expr::attr<std::string>("Tag") == "LogDataTag");
339 
340  logging::add_common_attributes(); // Add attributes like timestamp
341 
342  // Copy the gains file to the gains_directory with a new name containing the time info
343  std::string source_file = "./src/flightstack/params/control/mrac/gains_mrac.json";
344  std::stringstream target_ss;
345  target_ss << gains_directory << "/gains_mrac_" << std::put_time(std::localtime(&now_c), "%Y%m%d_%H%M%S") << ".json";
346  std::string target_file = target_ss.str();
347 
348  if (std::filesystem::exists(source_file)) {
349  std::cout << "GAINS FILE PRESENT" << std::endl;
350  std::filesystem::copy(source_file, target_file);
351  } else {
352  std::cout << "GAINS FILE NOT PRESENT" << std::endl;
353  }
354 
355  logInitializeHeaders();
356 }

208 /*
209  Computing Euler Angles from the unit quaternion following a 3-2-1 rotation sequence
210  The Euler Angles are ordered as: roll, pitch, yaw
211  For reference:
212  - https://github.com/PX4/PX4-Matrix/blob/master/matrix/Euler.hpp
213  - https://github.com/PX4/PX4-Matrix/blob/master/matrix/Dcm.hpp
214 */
215 Eigen::Vector3d VehicleState::quaternionToEulerAnglesRPY(const Eigen::Quaterniond& q) {
216  Eigen::Vector3d euler_angles_rpy;
217 
218  // Compute elements of Direction Cosine Matrix (DCM) from quaternion
219  double a = q.w();
220  double b = q.x();
221  double c = q.y();
222  double d = q.z();
223  double aa = a * a;
224  double ab = a * b;
225  double ac = a * c;
226  double ad = a * d;
227  double bb = b * b;
228  double bc = b * c;
229  double bd = b * d;
230  double cc = c * c;
231  double cd = c * d;
232  double dd = d * d;
233 
234  // Compute DCM elements
235  double dcm00 = aa + bb - cc - dd;
236  // double dcm01 = 2 * (bc - ad); // not needed by the algorithm
237  double dcm02 = 2 * (ac + bd);
238  double dcm10 = 2 * (bc + ad);
239  // double dcm11 = aa - bb + cc - dd; // not needed by the algorithm
240  double dcm12 = 2 * (cd - ab);
241  double dcm20 = 2 * (bd - ac);
242  double dcm21 = 2 * (ab + cd);
243  double dcm22 = aa - bb - cc + dd;
244 
245  // Compute Euler angles from DCM
246  euler_angles_rpy.y() = asin(-dcm20);
247 
248  if (fabs(euler_angles_rpy.y() - M_PI / 2) < 1.0e-3) {
249  euler_angles_rpy.x() = 0;
250  euler_angles_rpy.z() = atan2(dcm12, dcm02);
251  } else if (fabs(euler_angles_rpy.y() + M_PI / 2) < 1.0e-3) {
252  euler_angles_rpy.x() = 0;
253  euler_angles_rpy.z() = atan2(-dcm12, -dcm02);
254  } else {
255  euler_angles_rpy.x() = atan2(dcm21, dcm22);
256  euler_angles_rpy.z() = atan2(dcm10, dcm00);
257  }
258 
259  return euler_angles_rpy;
260 }

169 /*
170  Function to compute the variables used for the differentiator of the desired roll and pitch angles.
171  This function calculates the internal states and derivatives (first and second) of the desired roll and pitch
172  angles using the provided filter coefficients and the current state variables.
173 */
174 void PID::computeFilterDifferentiatorVariables(ControlInternalMembers& cim,
175  VehicleInfo& vehicle_info,
176  StateController& state_)
177 {
178  cim.internal_state_roll_des_filter = vehicle_info.A_filter_roll_des * state_.state_roll_des_filter
179  + vehicle_info.B_filter_roll_des * cim.roll_desired;
180 
181  cim.internal_state_pitch_des_filter = vehicle_info.A_filter_pitch_des * state_.state_pitch_des_filter
182  + vehicle_info.B_filter_pitch_des * cim.pitch_desired;
183 
184  cim.roll_desired_dot = vehicle_info.C_filter_roll_des * state_.state_roll_des_filter;
185  cim.pitch_desired_dot = vehicle_info.C_filter_pitch_des * state_.state_pitch_des_filter;
186 
187 
188  cim.internal_state_roll_dot_des_filter = vehicle_info.A_filter_roll_dot_des * state_.state_roll_dot_des_filter
189  + vehicle_info.B_filter_pitch_des * cim.roll_desired_dot;
190 
191  cim.internal_state_pitch_dot_des_filter = vehicle_info.A_filter_pitch_dot_des * state_.state_pitch_dot_des_filter
192  + vehicle_info.B_filter_pitch_dot_des * cim.pitch_desired_dot;
193 
194  cim.roll_desired_dot_dot = vehicle_info.C_filter_roll_dot_des * state_.state_roll_dot_des_filter;
195  cim.pitch_desired_dot_dot = vehicle_info.C_filter_pitch_dot_des * state_.state_pitch_dot_des_filter;
196 }

198 /*
199  OUTER LOOP CONTROLLER
200 */
201 void PID::computeOuterLoop(ControlInternalMembers& cim,
202  VehicleInfo& vehicle_info,
203  StateController& state_,
204  ControlReferences& cr,
205  GainsPID& gains_,
206  ControllerSpecificInternalMembers& csim_)
207 {
208  csim_.outer_loop_dynamic_inversion = - vehicle_info.mass * GRAVITATIONAL_ACCELERATION * this->e3_basis;
209  csim_.outer_loop_P = - gains_.KP_translational * cim.translational_position_error;
210  csim_.outer_loop_D = - gains_.KD_translational * (cr.velocity - cr.user_defined_velocity);
211  csim_.outer_loop_I = - gains_.KI_translational * state_.integral_translational_position_error;
212 
213  cim.mu_translational_raw = csim_.outer_loop_dynamic_inversion +
214  vehicle_info.mass * (
215  csim_.outer_loop_P
216  + csim_.outer_loop_D
217  + csim_.outer_loop_I
218  + cr.user_defined_acceleration
219  );
220 }

222 /*
223  INNER LOOP CONTROLLER
224 */
225 void PID::computeInnerLoop(ControlInternalMembers& cim,
226  VehicleInfo& vehicle_info,
227  StateController& state_,
228  ControlReferences& cr,
229  GainsPID& gains_,
230  ControllerSpecificInternalMembers& csim_)
231 {
232  csim_.inner_loop_dynamic_inversion = cr.angular_velocity.cross(vehicle_info.inertia_matrix * cr.angular_velocity);
233  csim_.inner_loop_P = - gains_.KP_rotational * cim.angular_error;
234  csim_.inner_loop_D = - gains_.KD_rotational * (cim.euler_angles_rpy_dot - cim.angular_position_desired_dot);
235  csim_.inner_loop_I = - gains_.KI_rotational * state_.integral_angular_error;
236 
237  cim.U2_U3_U4 = csim_.inner_loop_dynamic_inversion +
238  vehicle_info.inertia_matrix * (
239  csim_.inner_loop_P
240  + csim_.inner_loop_D
241  + csim_.inner_loop_I
242  + cim.angular_position_desired_dot_dot
243  );
244 }

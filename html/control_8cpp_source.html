176 // Function that computes the Jacobian matrix given roll and pitch angles
177 Eigen::Matrix3d Control::jacobianMatrix(const double& roll, const double& pitch)
178 {
179  Eigen::Matrix3d jacobian_matrix;
180 
181  jacobian_matrix(0, 0) = 1;
182  jacobian_matrix(0, 1) = 0;
183  jacobian_matrix(0, 2) = -std::sin(pitch);
184 
185  jacobian_matrix(1, 0) = 0;
186  jacobian_matrix(1, 1) = std::cos(roll);
187  jacobian_matrix(1, 2) = std::sin(roll) * std::cos(pitch);
188 
189  jacobian_matrix(2, 0) = 0;
190  jacobian_matrix(2, 1) = -std::sin(roll);
191  jacobian_matrix(2, 2) = std::cos(roll) * std::cos(pitch);
192 
193  return jacobian_matrix;
194 }

216  jacobian_matrix_dot(2, 2) = -std::sin(roll) * std::cos(pitch) * roll_dot
217  - std::cos(roll) * std::sin(pitch) * pitch_dot;

222 // Function that returns a 3x3 Jacobian matrix inverse given roll and pitch angles.
223 Eigen::Matrix3d Control::jacobianMatrixInverse(const double& roll, const double& pitch)
224 {
225  // Ensure pitch is not near +/- pi/2 to avoid division by zero
226  if (std::abs(std::cos(pitch)) < 1e-6) {
227  throw std::invalid_argument("Pitch value results in division by zero.");
228  }
229 
230  Eigen::Matrix3d jacobian_inverse;
231 
232  jacobian_inverse(0, 0) = 1;
233  jacobian_inverse(0, 1) = (std::sin(roll) * std::sin(pitch)) / std::cos(pitch);
234  jacobian_inverse(0, 2) = (std::cos(roll) * std::sin(pitch)) / std::cos(pitch);
235 
236  jacobian_inverse(1, 0) = 0;
237  jacobian_inverse(1, 1) = std::cos(roll);
238  jacobian_inverse(1, 2) = -std::sin(roll);
239 
240  jacobian_inverse(2, 0) = 0;
241  jacobian_inverse(2, 1) = std::sin(roll) / std::cos(pitch);
242  jacobian_inverse(2, 2) = std::cos(roll) / std::cos(pitch);
243 
244  return jacobian_inverse;
245 }

247 /*
248  Function that computes the 321 rotation matrix that transforms a vector from global to local coordinates.
249 
250  rotation_matrix_321_local_to_global = R3 * R2 * R1; // hence
251  rotation_matrix_321_global_to_local = R1_transpose * R2_transpose * R3_transpose;
252 */
253 Eigen::Matrix3d Control::rotationMatrix321GlobalToLocal(const double& roll, const double& pitch, const double& yaw)
254 {
255  Eigen::Matrix3d rotation_matrix_321_global_to_local;
256  Eigen::Matrix3d R1_transpose;
257  Eigen::Matrix3d R2_transpose;
258  Eigen::Matrix3d R3_transpose;
259 
260  R1_transpose = (Eigen::Matrix3d() <<
261  1, 0, 0,
262  0, std::cos(roll), std::sin(roll),
263  0, -std::sin(roll), std::cos(roll)
264  ).finished();
265 
266  R2_transpose = (Eigen::Matrix3d() <<
267  std::cos(pitch), 0, -std::sin(pitch),
268  0, 1, 0,
269  std::sin(pitch), 0, std::cos(pitch)
270  ).finished();
271 
272  R3_transpose = (Eigen::Matrix3d() <<
273  std::cos(yaw), std::sin(yaw), 0,
274  -std::sin(yaw), std::cos(yaw), 0,
275  0, 0, 1
276  ).finished();
277 
278  rotation_matrix_321_global_to_local = R1_transpose * R2_transpose * R3_transpose;
279 
280  return rotation_matrix_321_global_to_local;
281 }

379 /*
380  Function that computes:
381  the Total Thrust U1,
382  the desired roll angle RollDes,
383  the desired pitch angle PitchDes.
384  U1 will be fed to mixer matrix, while RollDes and PitchDes will be fed to the inner loop.
385 */
386 void Control::compute_U1_RollDes_PitchDes(ControlInternalMembers& cim)
387 {
388  // Convert the mu_translational from global to local coordinates
389  cim.mu_translational = cim.rotation_matrix_321_global_to_local * cim.mu_translational_raw;
390 
391  cim.U_control_inputs[0] = std::sqrt(
392  std::pow(cim.mu_translational[0], 2)
393  + std::pow(cim.mu_translational[1], 2)
394  + std::pow(cim.mu_translational[2], 2)
395  );
396 
397  double temporaryvar_roll_desired_1 = cim.mu_translational[1] / cim.U_control_inputs[0];
398  cim.roll_desired= std::atan2(
399  temporaryvar_roll_desired_1,
400  std::sqrt(1 - std::pow(temporaryvar_roll_desired_1, 2))
401  );
402 
403  cim.pitch_desired = std::atan2(-cim.mu_translational[0], -cim.mu_translational[2]);
404 }

470 /*
471  Function that makes the yaw angular error continuos considering the discontinuity/wrapping that occurs
472  at +/- pi for both the yaw and the user-defined yaw
473 */
474 double Control::makeYawAngularErrorContinuous(double yaw, double user_defined_yaw)
475 {
476  double continuos_error;
477 
478  user_defined_yaw = wrapAngleToMinusPiAndPi(user_defined_yaw);
479 
480  double raw_error = yaw - user_defined_yaw;
481 
482  if ( std::abs(raw_error) >= M_PI && std::abs(raw_error) < 2*M_PI && raw_error < 0 )
483  {
484  continuos_error = std::fmod(raw_error, M_PI) + M_PI;
485  }
486  else if ( std::abs(raw_error) >= M_PI && std::abs(raw_error) < 2*M_PI && raw_error > 0 )
487  {
488  continuos_error = std::fmod(raw_error, M_PI) - M_PI;
489  }
490  else
491  {
492  continuos_error = std::fmod(raw_error, M_PI);
493  }
494 
495  return continuos_error;
496 }

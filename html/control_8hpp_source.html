257  /*
258  Implementation of the smooth version of the dead-zone modification of MRAC through a
259  modulation function
260  For reference: E. Lavretsky, K. Wise, "Robust and Adaptive Control", Springer 2013, Sec. 11.2.1
261  PARAMETERS:
262  - delta: constant that must be 0 < delta < 1. delta characterizes the slope of the modulation function.
263  - e_0: constant that must be e_0 > 0. The dead-zone modification stops the adaptation process when
264  the norm of the tracking error becomes smaller than the prescribed value e_0.
265  */
266  template <typename T>
273  inline double deadZoneModulationFunction(const T& e_vector,
274  const double& delta,
275  const double& e_0)
276  {
277  return std::max(0.0, std::min(1.0, (e_vector.norm() - delta * e_0)/((1.0 - delta) * e_0)));
278  }
